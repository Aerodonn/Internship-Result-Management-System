CREATE TABLE UserLogin (
    UserID INT AUTO_INCREMENT PRIMARY KEY,
    Username varchar(255) NOT NULL,
    Password varchar(255) NOT NULL,
    SystemRole ENUM('Admin', 'Assessor') NOT NULL
);

CREATE TABLE Assessor (
	UserID Int PRIMARY KEY,
	FullName varchar(255) NOT NULL,
	PhoneNumber varchar(22), 
	EmailAddress varchar(128) UNIQUE NOT NULL,
	Organisation varchar(128),
	AssessorType ENUM('Lecturer', 'Industry Supervisor') NOT NULL,
	FOREIGN KEY (UserID) REFERENCES UserLogin(UserID)
);

CREATE TABLE Student (
	StudentID Int PRIMARY KEY AUTO_INCREMENT,
	StudentRegNumber varchar(20) NOT NULL UNIQUE,
	StudentName varchar(255) NOT NULL,
	EmailAddress varchar(255) UNIQUE NOT NULL,
	Programme varchar(255),
	EnrollmentDate DATE,
	AccountStatus ENUM('Active', 'Graduated', 'On-leave', 'Suspended') DEFAULT 'Active'
);

CREATE TABLE Internship(
	InternID Int PRIMARY KEY AUTO_INCREMENT,
	SupervisorID Int NOT NULL,
	LecturerID Int NOT NULL,
	StudentID Int NOT NULL,
	InternshipCompany varchar(255),
	StartDate DATE,
	EndDate DATE,
	ReportStatus ENUM('Drafting', 'In Progress', 'Suspended', 'Finalisation', 'Complete'),
	FOREIGN KEY (SupervisorID) REFERENCES Assessor(UserID),
	FOREIGN KEY (LecturerID) REFERENCES Assessor(UserID),
	FOREIGN KEY (StudentID) REFERENCES Student(StudentID)
);

CREATE TABLE InternshipReport(
	InternID Int PRIMARY KEY,
	taskScore DECIMAL(5,2) DEFAULT 0,
	safetyScore DECIMAL(5,2) DEFAULT 0,
	theoryScore DECIMAL(5,2) DEFAULT 0,
	presentScore DECIMAL(5,2) DEFAULT 0,
	clarityScore DECIMAL(5,2) DEFAULT 0,
	proj_mgmt_score DECIMAL(5,2) DEFAULT 0,
	time_mgmt_score DECIMAL(5,2) DEFAULT 0,
	comment TEXT,
	FOREIGN KEY (InternID) REFERENCES Internship(InternID)
);
